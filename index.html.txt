<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Financial Dashboard</title>
    
    <!-- React & ReactDOM -->
    <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    
    <!-- Babel (to interpret JSX) -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    
    <!-- Tailwind CSS -->
    <script src="https://cdn.tailwindcss.com"></script>

    <style>
        body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, "Helvetica Neue", Arial, sans-serif; }
    </style>
</head>
<body class="bg-slate-50 text-slate-900 pb-20">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect } = React;

        // --- Icons (SVG Components) ---
        const IconWallet = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M20 12V8H6a2 2 0 0 1-2-2c0-1.1.9-2 2-2h12v4"/><path d="M4 6v12a2 2 0 0 0 2 2h14v-4"/><path d="M18 12a2 2 0 0 0-2 2v4a2 2 0 0 0 2 2h4v-14"/></svg>
        );
        const IconCreditCard = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><rect width="20" height="14" x="2" y="5" rx="2"/><line x1="2" x2="22" y1="10" y2="10"/></svg>
        );
        const IconTrendingUp = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><polyline points="23 6 13.5 15.5 8.5 10.5 1 18"/><polyline points="17 6 23 6 23 12"/></svg>
        );
        const IconTrendingDown = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><polyline points="23 18 13.5 8.5 8.5 13.5 1 6"/><polyline points="17 18 23 18 23 12"/></svg>
        );
        const IconTrash2 = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M3 6h18"/><path d="M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6"/><path d="M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2"/><line x1="10" x2="10" y1="11" y2="17"/><line x1="14" x2="14" y1="11" y2="17"/></svg>
        );
        const IconPlus = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M5 12h14"/><path d="M12 5v14"/></svg>
        );
        const IconDollarSign = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><line x1="12" x2="12" y1="2" y2="22"/><path d="M17 5H9.5a3.5 3.5 0 0 0 0 7h5a3.5 3.5 0 0 1 0 7H6"/></svg>
        );
        const IconRefresh = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M3 12a9 9 0 0 1 9-9 9.75 9.75 0 0 1 6.74 2.74L21 8"/><path d="M21 3v5h-5"/><path d="M21 12a9 9 0 0 1-9 9 9.75 9.75 0 0 1-6.74-2.74L3 16"/><path d="M8 16H3v5"/></svg>
        );
        const IconSave = ({ className }) => (
            <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={className}><path d="M19 21H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11l5 5v11a2 2 0 0 1-2 2z"/><polyline points="17 21 17 13 7 13 7 21"/><polyline points="7 3 7 8 15 8"/></svg>
        );

        // --- Main App ---
        const FinancialTracker = () => {
            // Initial data
            const initialAssets = [
                { id: 1, name: 'HTX', amount: 0.92, color: 'bg-green-100 text-green-800' },
                { id: 2, name: 'Vantage', amount: 0.00, color: 'bg-purple-100 text-purple-800' },
                { id: 3, name: 'Touch n Go', amount: 29.88, color: 'bg-blue-100 text-blue-800' },
                { id: 4, name: 'RHB', amount: 56.24, color: 'bg-red-100 text-red-800' },
                { id: 5, name: 'Cash in LV wallet', amount: 65.00, color: 'bg-orange-100 text-orange-800' },
                { id: 6, name: 'AgroBiz', amount: 103.46, color: 'bg-gray-100 text-gray-800' },
                { id: 7, name: 'Agrobank', amount: 20.03, color: 'bg-gray-100 text-gray-800' },
                { id: 8, name: 'Bank Islam BE U', amount: 1717.29, color: 'bg-gray-200 text-gray-800' },
                { id: 9, name: 'CIMB', amount: 3000.00, color: 'bg-red-50 text-red-900' },
                { id: 10, name: 'KAF Digital Bank', amount: 30.00, color: 'bg-blue-50 text-blue-900' },
            ];

            // Load from LocalStorage or use default
            const [openingBalance, setOpeningBalance] = useState(() => {
                const saved = localStorage.getItem('openingBalance');
                return saved ? parseFloat(saved) : 227.33;
            });

            const [assets, setAssets] = useState(() => {
                const saved = localStorage.getItem('assets');
                return saved ? JSON.parse(saved) : initialAssets;
            });

            const [expenses, setExpenses] = useState(() => {
                const saved = localStorage.getItem('expenses');
                return saved ? JSON.parse(saved) : [
                    { id: 1, name: 'Lunch', amount: 15.50 },
                    { id: 2, name: 'Petrol', amount: 50.00 },
                ];
            });

            // Save to LocalStorage whenever state changes
            useEffect(() => {
                localStorage.setItem('openingBalance', openingBalance);
                localStorage.setItem('assets', JSON.stringify(assets));
                localStorage.setItem('expenses', JSON.stringify(expenses));
            }, [openingBalance, assets, expenses]);

            const [newAssetName, setNewAssetName] = useState('');
            const [newAssetAmount, setNewAssetAmount] = useState('');
            const [newExpenseName, setNewExpenseName] = useState('');
            const [newExpenseAmount, setNewExpenseAmount] = useState('');
            const [isEditingOpening, setIsEditingOpening] = useState(false);

            const totalAssets = assets.reduce((sum, asset) => sum + asset.amount, 0);
            const totalExpenses = expenses.reduce((sum, expense) => sum + expense.amount, 0);
            const profitOfTheDay = totalAssets - openingBalance; 

            const handleAssetChange = (id, newAmount) => {
                setAssets(assets.map(asset => 
                    asset.id === id ? { ...asset, amount: parseFloat(newAmount) || 0 } : asset
                ));
            };

            const addAsset = (e) => {
                e.preventDefault();
                if (!newAssetName) return;
                const newAsset = {
                    id: Date.now(),
                    name: newAssetName,
                    amount: parseFloat(newAssetAmount) || 0,
                    color: 'bg-slate-100 text-slate-800'
                };
                setAssets([...assets, newAsset]);
                setNewAssetName('');
                setNewAssetAmount('');
            };

            const removeAsset = (id) => setAssets(assets.filter(a => a.id !== id));

            const addExpense = (e) => {
                e.preventDefault();
                if (!newExpenseName) return;
                const newExpense = {
                    id: Date.now(),
                    name: newExpenseName,
                    amount: parseFloat(newExpenseAmount) || 0
                };
                setExpenses([...expenses, newExpense]);
                setNewExpenseName('');
                setNewExpenseAmount('');
            };

            const removeExpense = (id) => setExpenses(expenses.filter(e => e.id !== id));

            const formatCurrency = (val) => `RM ${val.toLocaleString('en-MY', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;

            // --- Components inside the main function for simplicity ---
            const StatCard = ({ title, amount, icon: Icon, colorClass, subText }) => (
                <div className={`p-6 rounded-2xl shadow-sm border border-slate-100 ${colorClass} transition-all duration-300`}>
                    <div className="flex justify-between items-start mb-2">
                        <h3 className="text-sm font-medium opacity-80">{title}</h3>
                        <Icon className="w-5 h-5 opacity-70" />
                    </div>
                    <div className="text-2xl font-bold tracking-tight">{formatCurrency(amount)}</div>
                    {subText && <div className="text-xs mt-1 opacity-70 font-medium">{subText}</div>}
                </div>
            );

            return (
                <div className="min-h-screen bg-slate-50 font-sans text-slate-900 pb-20">
                    {/* Header */}
                    <div className="bg-white border-b border-slate-200 sticky top-0 z-10 px-4 py-4 md:px-8 flex justify-between items-center shadow-sm">
                        <div className="flex items-center gap-2">
                            <div className="bg-indigo-600 p-2 rounded-lg text-white">
                                <IconWallet className="w-5 h-5" />
                            </div>
                            <h1 className="text-xl font-bold bg-clip-text text-transparent bg-gradient-to-r from-indigo-700 to-purple-600">
                                MyWealth
                            </h1>
                        </div>
                    </div>

                    <div className="max-w-4xl mx-auto p-4 md:p-8 space-y-8">
                        {/* Stats Grid */}
                        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                            <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200">
                                <div className="flex justify-between items-center mb-2">
                                    <span className="text-sm font-semibold text-slate-500 uppercase tracking-wider">Opening Balance</span>
                                    <button onClick={() => setIsEditingOpening(!isEditingOpening)} className="text-indigo-600">
                                        {isEditingOpening ? <IconSave className="w-4 h-4"/> : <IconRefresh className="w-4 h-4"/>}
                                    </button>
                                </div>
                                {isEditingOpening ? (
                                    <input 
                                        type="number" 
                                        className="w-full text-2xl font-bold border-b-2 border-indigo-500 focus:outline-none py-1"
                                        value={openingBalance}
                                        onChange={(e) => setOpeningBalance(parseFloat(e.target.value) || 0)}
                                    />
                                ) : (
                                    <div className="text-3xl font-bold text-slate-800">{formatCurrency(openingBalance)}</div>
                                )}
                            </div>

                            <StatCard title="Current Assets" amount={totalAssets} icon={IconDollarSign} colorClass="bg-white text-slate-800" subText={`${assets.length} Accounts`} />
                            
                            <StatCard 
                                title="Profit of the Day" 
                                amount={profitOfTheDay} 
                                icon={profitOfTheDay >= 0 ? IconTrendingUp : IconTrendingDown} 
                                colorClass={profitOfTheDay >= 0 ? 'bg-emerald-50 text-emerald-800 border-emerald-100' : 'bg-rose-50 text-rose-800 border-rose-100'} 
                                subText={profitOfTheDay >= 0 ? "+ Net Growth" : "Net Decrease"}
                            />
                        </div>

                        {/* Main Content */}
                        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
                            
                            {/* Assets Column */}
                            <div className="space-y-4">
                                <div className="flex items-center justify-between">
                                    <h2 className="text-lg font-bold flex items-center gap-2 text-slate-700">
                                        <IconCreditCard className="w-4 h-4" /> Liquid Assets
                                    </h2>
                                    <span className="text-xs bg-indigo-100 text-indigo-700 px-2 py-1 rounded-full font-bold">{formatCurrency(totalAssets)}</span>
                                </div>

                                <div className="bg-white rounded-2xl shadow-sm border border-slate-200 overflow-hidden">
                                    <div className="max-h-[500px] overflow-y-auto">
                                        <table className="w-full text-left border-collapse">
                                            <thead className="bg-slate-50 sticky top-0">
                                                <tr>
                                                    <th className="py-3 px-4 text-xs font-semibold text-slate-500 uppercase">Account</th>
                                                    <th className="py-3 px-4 text-xs font-semibold text-slate-500 uppercase text-right">Balance</th>
                                                    <th className="w-10"></th>
                                                </tr>
                                            </thead>
                                            <tbody className="divide-y divide-slate-100">
                                                {assets.map((asset) => (
                                                    <tr key={asset.id} className="group hover:bg-slate-50">
                                                        <td className="py-3 px-4">
                                                            <div className="flex items-center gap-3">
                                                                <div className={`w-2 h-8 rounded-full ${asset.color.split(' ')[0]}`}></div>
                                                                <span className="font-medium text-slate-700">{asset.name}</span>
                                                            </div>
                                                        </td>
                                                        <td className="py-3 px-4 text-right">
                                                            <input 
                                                                type="number"
                                                                value={asset.amount}
                                                                onChange={(e) => handleAssetChange(asset.id, e.target.value)}
                                                                className="w-24 text-right bg-transparent border-b border-transparent focus:border-indigo-500 focus:outline-none focus:bg-white font-mono text-slate-600 font-semibold"
                                                            />
                                                        </td>
                                                        <td className="py-3 px-2 text-center">
                                                            <button onClick={() => removeAsset(asset.id)} className="text-slate-300 hover:text-red-500">
                                                                <IconTrash2 className="w-4 h-4" />
                                                            </button>
                                                        </td>
                                                    </tr>
                                                ))}
                                            </tbody>
                                        </table>
                                    </div>
                                    <form onSubmit={addAsset} className="bg-slate-50 p-4 border-t border-slate-100 flex gap-2">
                                        <input type="text" placeholder="Name..." className="flex-1 px-3 py-2 rounded-lg border border-slate-300 text-sm" value={newAssetName} onChange={(e) => setNewAssetName(e.target.value)} />
                                        <input type="number" placeholder="0.00" className="w-20 px-3 py-2 rounded-lg border border-slate-300 text-sm" value={newAssetAmount} onChange={(e) => setNewAssetAmount(e.target.value)} />
                                        <button type="submit" className="bg-indigo-600 text-white p-2 rounded-lg"><IconPlus className="w-5 h-5" /></button>
                                    </form>
                                </div>
                            </div>

                            {/* Expenses Column */}
                            <div className="space-y-4">
                                <div className="flex items-center justify-between">
                                    <h2 className="text-lg font-bold flex items-center gap-2 text-slate-700">
                                        <IconTrendingDown className="w-4 h-4 text-red-500" /> Daily Expenses
                                    </h2>
                                    <span className="text-xs bg-red-100 text-red-700 px-2 py-1 rounded-full font-bold">{formatCurrency(totalExpenses)}</span>
                                </div>

                                <div className="bg-white rounded-2xl shadow-sm border border-slate-200 overflow-hidden">
                                    <ul className="divide-y divide-slate-100">
                                        {expenses.map((expense) => (
                                            <li key={expense.id} className="p-4 flex justify-between items-center group hover:bg-slate-50">
                                                <div className="flex items-center gap-3">
                                                    <div className="bg-red-100 p-2 rounded-full text-red-600"><IconDollarSign className="w-3 h-3" /></div>
                                                    <span className="font-medium text-slate-700">{expense.name}</span>
                                                </div>
                                                <div className="flex items-center gap-4">
                                                    <span className="font-mono font-semibold text-slate-600">{formatCurrency(expense.amount)}</span>
                                                    <button onClick={() => removeExpense(expense.id)} className="text-slate-300 hover:text-red-500"><IconTrash2 className="w-4 h-4" /></button>
                                                </div>
                                            </li>
                                        ))}
                                    </ul>
                                    <form onSubmit={addExpense} className="bg-slate-50 p-4 border-t border-slate-100 flex gap-2">
                                        <input type="text" placeholder="Expense..." className="flex-1 px-3 py-2 rounded-lg border border-slate-300 text-sm" value={newExpenseName} onChange={(e) => setNewExpenseName(e.target.value)} />
                                        <input type="number" placeholder="0.00" className="w-20 px-3 py-2 rounded-lg border border-slate-300 text-sm" value={newExpenseAmount} onChange={(e) => setNewExpenseAmount(e.target.value)} />
                                        <button type="submit" className="bg-red-500 text-white p-2 rounded-lg"><IconPlus className="w-5 h-5" /></button>
                                    </form>
                                </div>
                            </div>
                        </div>
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<FinancialTracker />);
    </script>
</body>
</html>
