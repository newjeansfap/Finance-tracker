<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>Financial Tracker</title>
    
    <!-- React & ReactDOM -->
    <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    
    <!-- Babel -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    
    <!-- Tailwind CSS -->
    <script src="https://cdn.tailwindcss.com"></script>

    <style>
        body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, Helvetica, Arial, sans-serif; -webkit-tap-highlight-color: transparent; }
        input, select { font-size: 16px !important; } /* Stop iPhone Zoom */
    </style>
</head>
<body class="bg-slate-50 text-slate-900 pb-20">
    <div id="root" class="p-10 text-center text-slate-500 font-bold">
        Loading App...
    </div>

    <script type="text/babel">
        try {
            const { useState, useEffect, useMemo } = React;

            // --- Icons ---
            const IconWallet = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M20 12V8H6a2 2 0 0 1-2-2c0-1.1.9-2 2-2h12v4"/><path d="M4 6v12a2 2 0 0 0 2 2h14v-4"/><path d="M18 12a2 2 0 0 0-2 2v4a2 2 0 0 0 2 2h4v-14"/></svg>;
            const IconCalendar = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><rect x="3" y="4" width="18" height="18" rx="2" ry="2"></rect><line x1="16" y1="2" x2="16" y2="6"></line><line x1="8" y1="2" x2="8" y2="6"></line><line x1="3" y1="10" x2="21" y2="10"></line></svg>;
            const IconTrendingUp = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><polyline points="23 6 13.5 15.5 8.5 10.5 1 18"/><polyline points="17 6 23 6 23 12"/></svg>;
            const IconTrendingDown = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><polyline points="23 18 13.5 8.5 8.5 13.5 1 6"/><polyline points="17 18 23 18 23 12"/></svg>;
            const IconTrash2 = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M3 6h18"/><path d="M19 6v14c0 1-1 2-2 2H7c-1 0-2-1-2-2V6"/><path d="M8 6V4c0-1 1-2 2-2h4c1 0 2 1 2 2v2"/><line x1="10" x2="10" y1="11" y2="17"/><line x1="14" x2="14" y1="11" y2="17"/></svg>;
            const IconPlus = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M5 12h14"/><path d="M12 5v14"/></svg>;
            const IconList = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><line x1="8" y1="6" x2="21" y2="6"></line><line x1="8" y1="12" x2="21" y2="12"></line><line x1="8" y1="18" x2="21" y2="18"></line><line x1="3" y1="6" x2="3.01" y2="6"></line><line x1="3" y1="12" x2="3.01" y2="12"></line><line x1="3" y1="18" x2="3.01" y2="18"></line></svg>;
            const IconHome = (p) => <svg {...p} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M3 9l9-7 9 7v11a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2z"></path><polyline points="9 22 9 12 15 12 15 22"></polyline></svg>;

            const FinancialTracker = () => {
                // --- State ---
                const [view, setView] = useState('daily'); // 'daily' or 'monthly'
                const [selectedDate, setSelectedDate] = useState(new Date().toISOString().split('T')[0]); // YYYY-MM-DD
                
                // Initial Assets
                const initialAssets = [
                    { id: 1, name: 'HTX', amount: 0.92, color: 'bg-green-100 text-green-800' },
                    { id: 2, name: 'Vantage', amount: 0.00, color: 'bg-purple-100 text-purple-800' },
                    { id: 3, name: 'Touch n Go', amount: 29.88, color: 'bg-blue-100 text-blue-800' },
                    { id: 4, name: 'RHB', amount: 56.24, color: 'bg-red-100 text-red-800' },
                    { id: 5, name: 'Cash in LV wallet', amount: 65.00, color: 'bg-orange-100 text-orange-800' },
                    { id: 6, name: 'AgroBiz', amount: 103.46, color: 'bg-gray-100 text-gray-800' },
                    { id: 7, name: 'Agrobank', amount: 20.03, color: 'bg-gray-100 text-gray-800' },
                    { id: 8, name: 'Bank Islam BE U', amount: 1717.29, color: 'bg-gray-200 text-gray-800' },
                    { id: 9, name: 'CIMB', amount: 3000.00, color: 'bg-red-50 text-red-900' },
                    { id: 10, name: 'KAF Digital Bank', amount: 30.00, color: 'bg-blue-50 text-blue-900' },
                ];

                // Load Assets
                const [assets, setAssets] = useState(() => {
                    try {
                        const saved = localStorage.getItem('assets_v2');
                        return saved ? JSON.parse(saved) : initialAssets;
                    } catch(e) { return initialAssets; }
                });

                // Load Opening Balance
                const [openingBalance, setOpeningBalance] = useState(() => {
                    try {
                        const saved = localStorage.getItem('openingBalance_v2');
                        return saved ? parseFloat(saved) : 227.33;
                    } catch(e) { return 227.33; }
                });

                // Load Expenses (With Date Support)
                const [allExpenses, setAllExpenses] = useState(() => {
                    try {
                        const saved = localStorage.getItem('expenses_v2');
                        let parsed = saved ? JSON.parse(saved) : [];
                        // Migration: If old data has no date, assign today
                        const today = new Date().toISOString().split('T')[0];
                        parsed = parsed.map(e => e.date ? e : { ...e, date: today });
                        return parsed;
                    } catch(e) { return []; }
                });

                // --- Persistence ---
                useEffect(() => {
                    localStorage.setItem('assets_v2', JSON.stringify(assets));
                    localStorage.setItem('openingBalance_v2', openingBalance);
                    localStorage.setItem('expenses_v2', JSON.stringify(allExpenses));
                }, [assets, openingBalance, allExpenses]);

                // --- Inputs State ---
                const [newAssetName, setNewAssetName] = useState('');
                const [newAssetAmount, setNewAssetAmount] = useState('');
                const [newExpenseName, setNewExpenseName] = useState('');
                const [newExpenseAmount, setNewExpenseAmount] = useState('');
                const [isEditingOpening, setIsEditingOpening] = useState(false);

                // --- Computed Data ---
                const currentDayExpenses = useMemo(() => {
                    return allExpenses.filter(e => e.date === selectedDate);
                }, [allExpenses, selectedDate]);

                const totalDailyExpenses = currentDayExpenses.reduce((sum, e) => sum + e.amount, 0);
                const totalAssets = assets.reduce((sum, a) => sum + a.amount, 0);
                const profitOfTheDay = totalAssets - openingBalance;

                // Monthly Calculation
                const selectedMonth = selectedDate.substring(0, 7); // "2023-10"
                const monthExpenses = useMemo(() => {
                    return allExpenses.filter(e => e.date.startsWith(selectedMonth));
                }, [allExpenses, selectedMonth]);
                const totalMonthExpenses = monthExpenses.reduce((sum, e) => sum + e.amount, 0);

                // --- Handlers ---
                const handleAssetChange = (id, newAmount) => {
                    setAssets(assets.map(asset => 
                        asset.id === id ? { ...asset, amount: parseFloat(newAmount) || 0 } : asset
                    ));
                };

                const addAsset = (e) => {
                    e.preventDefault();
                    if (!newAssetName) return;
                    setAssets([...assets, {
                        id: Date.now(),
                        name: newAssetName,
                        amount: parseFloat(newAssetAmount) || 0,
                        color: 'bg-slate-100 text-slate-800'
                    }]);
                    setNewAssetName('');
                    setNewAssetAmount('');
                };

                const addExpense = (e) => {
                    e.preventDefault();
                    if (!newExpenseName) return;
                    setAllExpenses([...allExpenses, {
                        id: Date.now(),
                        name: newExpenseName,
                        amount: parseFloat(newExpenseAmount) || 0,
                        date: selectedDate // Saves with the currently selected date
                    }]);
                    setNewExpenseName('');
                    setNewExpenseAmount('');
                };

                const removeExpense = (id) => {
                    if(confirm("Delete this expense?")) {
                        setAllExpenses(allExpenses.filter(e => e.id !== id));
                    }
                };

                const removeAsset = (id) => {
                    if(confirm("Remove this account?")) {
                        setAssets(assets.filter(a => a.id !== id));
                    }
                };

                const formatCurrency = (val) => `RM ${val.toLocaleString('en-MY', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`;

                // --- Views ---

                const DailyView = () => (
                    <div className="space-y-6 animate-fade-in">
                        {/* Summary Cards */}
                        <div className="bg-white p-5 rounded-xl shadow-sm border border-slate-200">
                            <div className="flex justify-between items-center mb-2">
                                <span className="text-xs font-bold text-slate-400 uppercase tracking-wider">Opening Balance (Today)</span>
                                <button onClick={() => setIsEditingOpening(!isEditingOpening)} className="text-indigo-600 text-xs font-bold">
                                    {isEditingOpening ? 'SAVE' : 'EDIT'}
                                </button>
                            </div>
                            {isEditingOpening ? (
                                <input 
                                    type="number" 
                                    className="w-full text-3xl font-bold border-b-2 border-indigo-500 focus:outline-none py-1"
                                    value={openingBalance}
                                    onChange={(e) => setOpeningBalance(parseFloat(e.target.value) || 0)}
                                />
                            ) : (
                                <div className="text-3xl font-bold text-slate-800">{formatCurrency(openingBalance)}</div>
                            )}
                        </div>

                        <div className="grid grid-cols-2 gap-4">
                            <div className="p-4 bg-white rounded-xl shadow-sm border border-slate-100">
                                <div className="text-xs text-slate-500 font-bold uppercase mb-1">Profit Today</div>
                                <div className={`text-xl font-bold ${profitOfTheDay >= 0 ? 'text-emerald-600' : 'text-rose-600'}`}>
                                    {profitOfTheDay >= 0 ? '+' : ''}{formatCurrency(profitOfTheDay)}
                                </div>
                            </div>
                            <div className="p-4 bg-white rounded-xl shadow-sm border border-slate-100">
                                <div className="text-xs text-slate-500 font-bold uppercase mb-1">Expenses Today</div>
                                <div className="text-xl font-bold text-slate-800">{formatCurrency(totalDailyExpenses)}</div>
                            </div>
                        </div>

                        {/* Two Columns for Desktop, Stacked for Mobile */}
                        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
                            
                            {/* Assets Section */}
                            <div className="space-y-3">
                                <h2 className="text-sm font-bold text-slate-500 uppercase flex items-center gap-2">
                                    <IconWallet className="w-4 h-4" /> Assets
                                </h2>
                                <div className="bg-white rounded-xl shadow-sm border border-slate-200 overflow-hidden">
                                    <div className="max-h-[40vh] overflow-y-auto">
                                        <table className="w-full text-left">
                                            <tbody className="divide-y divide-slate-100">
                                                {assets.map((asset) => (
                                                    <tr key={asset.id}>
                                                        <td className="py-3 px-4">
                                                            <div className="flex items-center gap-2">
                                                                <div className={`w-1.5 h-8 rounded-full ${asset.color.split(' ')[0]}`}></div>
                                                                <span className="font-medium text-slate-700 text-sm">{asset.name}</span>
                                                            </div>
                                                        </td>
                                                        <td className="py-3 px-4 text-right">
                                                            <input 
                                                                type="number"
                                                                value={asset.amount}
                                                                onChange={(e) => handleAssetChange(asset.id, e.target.value)}
                                                                className="w-24 text-right bg-slate-50 rounded p-1 font-mono text-slate-700 font-semibold"
                                                            />
                                                        </td>
                                                    </tr>
                                                ))}
                                            </tbody>
                                        </table>
                                    </div>
                                    {/* Add Asset */}
                                    <form onSubmit={addAsset} className="bg-slate-50 p-3 border-t border-slate-100 flex gap-2">
                                        <input type="text" placeholder="New Account..." className="flex-1 px-3 py-2 rounded-lg border border-slate-300 bg-white" value={newAssetName} onChange={(e) => setNewAssetName(e.target.value)} />
                                        <input type="number" placeholder="0.00" className="w-20 px-3 py-2 rounded-lg border border-slate-300 bg-white" value={newAssetAmount} onChange={(e) => setNewAssetAmount(e.target.value)} />
                                        <button type="submit" className="bg-slate-200 text-slate-600 p-2 rounded-lg"><IconPlus className="w-5 h-5" /></button>
                                    </form>
                                </div>
                            </div>

                            {/* Expenses Section */}
                            <div className="space-y-3">
                                <h2 className="text-sm font-bold text-slate-500 uppercase flex items-center gap-2">
                                    <IconTrendingDown className="w-4 h-4" /> Expenses ({selectedDate})
                                </h2>
                                <div className="bg-white rounded-xl shadow-sm border border-slate-200 overflow-hidden">
                                    <ul className="divide-y divide-slate-100 max-h-[40vh] overflow-y-auto">
                                        {currentDayExpenses.length === 0 && (
                                            <li className="p-6 text-center text-slate-400 italic text-sm">No expenses for this date.</li>
                                        )}
                                        {currentDayExpenses.map((expense) => (
                                            <li key={expense.id} className="p-3 flex justify-between items-center">
                                                <span className="font-medium text-slate-700 text-sm pl-2">{expense.name}</span>
                                                <div className="flex items-center gap-3">
                                                    <span className="font-mono font-bold text-slate-600">{formatCurrency(expense.amount)}</span>
                                                    <button onClick={() => removeExpense(expense.id)} className="text-slate-300 hover:text-red-500"><IconTrash2 className="w-4 h-4" /></button>
                                                </div>
                                            </li>
                                        ))}
                                    </ul>
                                    {/* Add Expense */}
                                    <form onSubmit={addExpense} className="bg-slate-50 p-3 border-t border-slate-100 flex gap-2">
                                        <input type="text" placeholder="Expense..." className="flex-1 px-3 py-2 rounded-lg border border-slate-300 bg-white" value={newExpenseName} onChange={(e) => setNewExpenseName(e.target.value)} />
                                        <input type="number" placeholder="0.00" className="w-20 px-3 py-2 rounded-lg border border-slate-300 bg-white" value={newExpenseAmount} onChange={(e) => setNewExpenseAmount(e.target.value)} />
                                        <button type="submit" className="bg-red-500 text-white p-2 rounded-lg shadow-md shadow-red-200"><IconPlus className="w-5 h-5" /></button>
                                    </form>
                                </div>
                            </div>
                        </div>
                    </div>
                );

                const MonthlyView = () => {
                    // Group expenses by date
                    const grouped = monthExpenses.reduce((groups, expense) => {
                        const date = expense.date;
                        if (!groups[date]) groups[date] = [];
                        groups[date].push(expense);
                        return groups;
                    }, {});

                    // Sort dates (newest first)
                    const sortedDates = Object.keys(grouped).sort((a, b) => new Date(b) - new Date(a));

                    return (
                        <div className="space-y-6 animate-fade-in">
                            <div className="bg-gradient-to-r from-indigo-500 to-purple-600 p-6 rounded-xl text-white shadow-lg">
                                <div className="text-indigo-100 text-sm font-medium mb-1">Total Spent in {new Date(selectedDate).toLocaleString('default', { month: 'long', year: 'numeric' })}</div>
                                <div className="text-4xl font-bold">{formatCurrency(totalMonthExpenses)}</div>
                            </div>

                            <div className="space-y-4">
                                {sortedDates.length === 0 ? (
                                    <div className="text-center py-10 text-slate-400">No expenses recorded for this month.</div>
                                ) : (
                                    sortedDates.map(date => (
                                        <div key={date} className="bg-white rounded-xl shadow-sm border border-slate-200 overflow-hidden">
                                            <div className="bg-slate-50 px-4 py-2 border-b border-slate-100 flex justify-between items-center">
                                                <span className="font-bold text-slate-600 text-sm">{new Date(date).toLocaleDateString(undefined, { weekday: 'short', month: 'short', day: 'numeric' })}</span>
                                                <span className="text-xs font-bold bg-slate-200 text-slate-600 px-2 py-1 rounded-full">
                                                    {formatCurrency(grouped[date].reduce((s, e) => s + e.amount, 0))}
                                                </span>
                                            </div>
                                            <div className="divide-y divide-slate-50">
                                                {grouped[date].map(expense => (
                                                    <div key={expense.id} className="p-3 flex justify-between items-center hover:bg-slate-50">
                                                        <span className="text-slate-700 text-sm pl-2">{expense.name}</span>
                                                        <span className="font-mono text-slate-500 text-sm">{formatCurrency(expense.amount)}</span>
                                                    </div>
                                                ))}
                                            </div>
                                        </div>
                                    ))
                                )}
                            </div>
                        </div>
                    );
                };

                return (
                    <div className="min-h-screen bg-slate-50 font-sans text-slate-900 pb-24">
                        {/* Header */}
                        <div className="bg-white border-b border-slate-200 sticky top-0 z-20 px-4 py-3 flex justify-between items-center shadow-sm">
                            <h1 className="text-lg font-bold bg-clip-text text-transparent bg-gradient-to-r from-indigo-700 to-purple-600">
                                MyWealth
                            </h1>
                            <div className="flex items-center gap-2">
                                <IconCalendar className="w-5 h-5 text-slate-400" />
                                <input 
                                    type="date" 
                                    value={selectedDate} 
                                    onChange={(e) => setSelectedDate(e.target.value)}
                                    className="bg-slate-100 text-slate-700 font-bold rounded-lg px-2 py-1 text-sm outline-none focus:ring-2 focus:ring-indigo-500"
                                />
                            </div>
                        </div>

                        <div className="max-w-2xl mx-auto p-4">
                            {view === 'daily' ? <DailyView /> : <MonthlyView />}
                        </div>

                        {/* Bottom Navigation */}
                        <div className="fixed bottom-0 left-0 right-0 bg-white border-t border-slate-200 p-2 pb-6 flex justify-around z-30 shadow-[0_-4px_6px_-1px_rgba(0,0,0,0.05)]">
                            <button 
                                onClick={() => setView('daily')}
                                className={`flex flex-col items-center gap-1 px-6 py-2 rounded-xl transition-all ${view === 'daily' ? 'text-indigo-600 bg-indigo-50' : 'text-slate-400'}`}
                            >
                                <IconHome className="w-6 h-6" />
                                <span className="text-[10px] font-bold uppercase tracking-wide">Daily</span>
                            </button>
                            <button 
                                onClick={() => setView('monthly')}
                                className={`flex flex-col items-center gap-1 px-6 py-2 rounded-xl transition-all ${view === 'monthly' ? 'text-indigo-600 bg-indigo-50' : 'text-slate-400'}`}
                            >
                                <IconList className="w-6 h-6" />
                                <span className="text-[10px] font-bold uppercase tracking-wide">Monthly Report</span>
                            </button>
                        </div>
                    </div>
                );
            };

            const root = ReactDOM.createRoot(document.getElementById('root'));
            root.render(<FinancialTracker />);

        } catch (error) {
            document.getElementById('root').innerHTML = `<div style="color:red; padding:20px;">Error: ${error.message}</div>`;
        }
    </script>
</body>
</html>
